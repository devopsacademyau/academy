     {
         "Version": "2012-10-17",
         "Statement": [
             {
                 "Effect": "Allow",
                 "Action": [
                     "s3:Get*",
                     "s3:List*"
                 ],
                 "Resource": [
                     "arn:aws:s3:::mail2kvanitha",
                     "arn:aws:s3:::mail2kvanitha/*"
                 ]
             }
         ]
     }
